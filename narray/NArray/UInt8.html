static VALUE
nary_uint8_s_cast(VALUE type, VALUE obj)
{
    VALUE v;
    narray_t *na;
    dtype x;

    if (CLASS_OF(obj)==cT) {
        return obj;
    }
    if (RTEST(rb_obj_is_kind_of(obj,rb_cNumeric))) {
        x = m_num_to_data(obj);
        return nary_uint8_new_dim0(x);
    }
    if (RTEST(rb_obj_is_kind_of(obj,rb_cArray))) {
        return nary_uint8_cast_array(obj);
    }
    if (IsNArray(obj)) {
        GetNArray(obj,na);
        v = rb_narray_new(cT, NA_NDIM(na), NA_SHAPE(na));
        if (NA_SIZE(na)>0) {
            na_alloc_data(v);
            nary_uint8_store(v,obj);
        }
        return v;
    }
    rb_raise(nary_eCastError,"cannot cast to %s",rb_class2name(type));
    return Qnil;
}

VALUE
nary_uint8_extract(VALUE self)
{
    volatile VALUE v;
    char *ptr;
    narray_t *na;
    GetNArray(self,na);

    if (na->ndim==0) {
        ptr = na_get_pointer_for_read(self) + na_get_offset(self);
        v = m_extract(ptr);
        na_release_lock(self);
        return v;
    }
    return self;
}

nary_uint8_store(VALUE self, VALUE obj)
{
    VALUE r;

    
    if (FIXNUM_P(obj) || TYPE(obj)==T_FLOAT || TYPE(obj)==T_BIGNUM || rb_obj_is_kind_of(obj,rb_cComplex)) {
        nary_uint8_store_numeric(self,obj);
        return self;
    }
    
    if (TYPE(obj)==T_ARRAY) {
        nary_uint8_store_array(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cDFloat)) {
        nary_uint8_store_dfloat(self,obj);
        return self;
    }

        nary_uint8_store_sfloat(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cInt64)) {
        nary_uint8_store_int64(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cInt32)) {
        nary_uint8_store_int32(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cInt16)) {
        nary_uint8_store_int16(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cInt8)) {
        nary_uint8_store_int8(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cUInt64)) {
        nary_uint8_store_uint64(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cUInt32)) {
        nary_uint8_store_uint32(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cUInt16)) {
        nary_uint8_store_uint16(self,obj);
        return self;
    }
    
    if (rb_obj_is_kind_of(obj,cUInt8)) {
        nary_uint8_store_uint8(self,obj);
        return self;
    }
    

    if (IsNArray(obj)) {
        r = rb_funcall(obj, rb_intern("coerce_cast"), 1, cT);
        if (CLASS_OF(r)==cT) {
            nary_uint8_store(self,r);
            return self;
        }

    rb_raise(nary_eCastError, "unknown conversion from %s to %s",
